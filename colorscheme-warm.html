python# Data processing pipeline
from dataclasses import dataclass
from typing import Optional

@dataclass
class Record:
    name: str
    value: float
    tag: Optional[str] = None

def process(records: list[Record]) -> dict:
    result = {}
    for rec in records:
        if rec.value > 0:
            result[rec.name] = rec.value
    return result
